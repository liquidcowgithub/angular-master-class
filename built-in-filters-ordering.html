Built-In Filters: orderBy

HTML

                
<h3>Athletes & Countries & Medals</h3>
<table>
    <thead style="cursor: pointer;">
        <th ng-click="$ctrl.orderTableBy('name');">Name</th>
        <th ng-click="$ctrl.orderTableBy('country');">Country</th>
        <th ng-click="$ctrl.orderTableBy('goldMedals');">Gold Medals</th>
    </thead>
    <tbody>
        <tr ng-repeat="athlete in $ctrl.athletes | orderBy:$ctrl.column:$ctrl.reverse">
            <td>{{athlete.name}}</td>
            <td>{{athlete.country}}</td>
            <td>{{athlete.goldMedals}}</td>
        </tr>
    </tbody>
</table>

JavaScript

                
var exampleModule = angular.module('exampleModule',[]);
exampleModule.controller('exampleController', function() {
    var $ctrl = this;
    $ctrl.column = 'name';
    $ctrl.reverse = true;
    $ctrl.orderTableBy = function orderTableBy(columnName){
        $ctrl.column = columnName;
        $ctrl.reverse = !$ctrl.reverse;
    };
    $ctrl.athletes = [{ name: "Paavo Nurmi", country: "Finland", goldMedals: 9 }, ...];
});
