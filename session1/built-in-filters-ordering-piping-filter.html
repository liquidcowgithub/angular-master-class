Built-In Filters: Filter

HTML

                
<h3>Athletes & Countries & Medals</h3>
<label>Search: <input ng-model="$ctrl.resultsSearch"></input></label>
<label>Limit results to: <input ng-model="$ctrl.resultsLimit"></input></label>
<table>
    <thead style="cursor: pointer;">
        <th ng-click="$ctrl.orderTableBy('name');">Name</th>
        ...
    </thead>
    <tbody>
        <tr ng-repeat="athlete in $ctrl.athletes
            | filter:$ctrl.resultsSearch
            | orderBy:$ctrl.column:$ctrl.reverse
            | limitTo:$ctrl.resultsLimit ">
            <td>{{athlete.name}}</td>
            ...
        </tr>
    </tbody>
</table>

JavaScript

                
var exampleModule = angular.module('exampleModule',[]);
exampleModule.controller('exampleController', function() {
    var $ctrl = this;
    $ctrl.column = 'name';
    $ctrl.reverse = true;
    $ctrl.resultsLimit = 3;
    $ctrl.orderTableBy = function orderTableBy(columnName){
        $ctrl.column = columnName;
        $ctrl.reverse = !$ctrl.reverse;
    };
    $ctrl.athletes = [{ name: "Paavo Nurmi", country: "Finland", goldMedals: 9 }, ...];
});
