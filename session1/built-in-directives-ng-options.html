Built-In Directives: ngOptions



HTML

                    
<p>selected athlete is : {{selectedAthlete.name}}</p>
<p>total medals of selected athlete is :
{{selectedAthlete.medals}} </p>

<select ng-model="selectedAthlete"
ng-options="athlete.name for athlete in athletes">
</select>

JavaScript

                
var exampleModule = angular.module('exampleModule',[]);
exampleModule.controller('exampleController', function() {
  $scope.athletes = [
    {name: 'Usain Bolt', medals: 30 },
    { name: 'Michael Phelps', medals: 27 },
    { name: 'Tom Daley', medals: 50 }
  ];
});

Outcome

              
Selected athlete is : {{$ctrl.selectedAthlete.name}}

              
Total medals of selected athlete is : {{$ctrl.selectedAthlete.medals}}
