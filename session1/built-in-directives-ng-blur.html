Built-In Directives: ngBlur



HTML

                    
<input type="text" ng-focus="focus()" ng-blur="blur()">
<h3>{{focusText}}</h3>

JavaScript

                
var exampleModule = angular.module('exampleModule',[]);
exampleModule.controller('exampleController', function() {
  var $ctrl = this;
  $ctrl.focusText;

  $ctrl.focus = function(){
    $ctrl.focusText = 'Focused :)';
  }

  $ctrl.blur = function(){
    $ctrl.focusText = 'Not focused :(';
  }
});

Outcome

              
              
{{$ctrl.focusText}}
