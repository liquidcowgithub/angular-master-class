Built-In Directives: ngChange



HTML

                    
<input type="text" ng-change="changeFunc()"
ng-model="changeValue" />
<p>Usain will win {{count}} gold medals.</p>

JavaScript

                
var exampleModule = angular.module('exampleModule',[]);
exampleModule.controller('exampleController', function() {
  $scope.count = 0;

  $scope.changeFunc = function() {
      $scope.count++;
  };
});

Outcome

              
              

              
Usain will win {{$ctrl.count}} gold medals.
