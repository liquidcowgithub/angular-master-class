angular
    .module('myRootModule', ['ui.router'])
    .config(function($stateProvider, $urlRouterProvider) {
        $urlRouterProvider.otherwise("/state1");

        $stateProvider
            .state('state1', {
                url: "/state1",
                templateUrl: "myModule/templates/state1.html"
            })
            .state('state2', {
                url: "/state2",
                templateUrl: "myModule/templates/state2.html"
            });
});
