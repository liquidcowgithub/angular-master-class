Built-In Directives: ngRepeat

HTML

                
                    
                    <h3>Kittens & Reactions</h3>
                    <div ng-repeat="kitten in $ctrl.kittens">
                        {{kitten.name}} -> {{kitten.cuteness}}
                    </div>

JavaScript

                
var exampleModule = angular.module('exampleModule',[]);
exampleModule.controller('exampleController', function() {

    var $ctrl = this;

    $ctrl.kittens = [{
        name: "Mittens",
        cuteness: "meh -_-"
    },
    {
        name: "Snowball",
        cuteness: ":D"
    },
    {
        name: "Daisy",
        cuteness: "(♥_♥)"
    }];
});

Outcome

                
Kittens & Reactions

                

                    {{kitten.name}} -> {{kitten.cuteness}}
