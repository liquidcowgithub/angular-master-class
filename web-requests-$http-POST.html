JavaScript

                
var exampleModule = angular.module('exampleModule',[]);
exampleModule.controller('exampleController', function($http) {
    var $ctrl = this;
    $ctrl.newSupplier = {};
    $ctrl.postNewSupplier = postNewSupplier;

    getSuppliers();

    function postNewSupplier(){
        $http({
            method: 'POST',
            url: 'http://frontendshowcase.azurewebsites.net/api/suppliers',
            data: $ctrl.newSupplier
        })
        .then(getSuppliers);
    }

    function getSuppliers(){
        $http({
            method: 'GET',
            url: 'http://frontendshowcase.azurewebsites.net/api/suppliers'
        })
        .then(function(response){
            $ctrl.suppliers = response.data;
        });
    }
});
